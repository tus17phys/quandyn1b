		\subsubsection{光子 -photon-}

			振動数$\nu$，波長$\lambda$の電磁波は，
			「エネルギー
			\begin{equation}
				E = h \nu \label{ehnu}
			\end{equation}
			運動量
			\begin{equation}
				 p = \frac{h}{\lambda} \label{phl}
			\end{equation}
			の光子の集合体」
			であるとした．\par
			光子について
			\begin{equation}
				\lambda = \frac{h}{\nu} \label{lhnu}
			\end{equation}
			より(\ref{phl})は，
			\begin{equation}
				p = h\frac{\nu}{c} \label{phnu}
			\end{equation}

			(\ref{ehnu})，(\ref{phnu})より，
			\begin{equation}
				E = {c}{p} ←光子について成り立つ
			\end{equation}
		\subsubsection{(注)相対論}
				総エネルギーの式
				\begin{equation}
					E^2 = {m_0}^2c^4 + p^2 c^2
				\end{equation}

				（$m_0$：静止質量，$p$：$mv$）
				運動質量$m$は
				\begin{equation}
					m\sqrt{1 - \frac{v^2}{c^2}} = m_0
				\end{equation}
				光子について
				$v = c$より
				$m_0 = 0$
				よって
				$E = pc$
				で一致している．

		\subsubsection{物質波}
			de Broglieは光の粒子性を逆に読み，物質が波動を伴うとした．
			エネルギー$E$，運動量$p$を持つ粒子は，
			\begin{eqnarray}
				\nu = \frac{E}{h} \\
				\lambda = \frac{h}{p}
			\end{eqnarray}
			の波動を伴う．（$\lambda$はde Broglie波長）\par
			粒子について
			\begin{equation}
				\lambda = \frac{h}{p} = \frac{h}{mv}
			\end{equation}
			(光子の場合$\lambda = c/\nu$) \\
			粒子の場合$\lambda = v/\nu$.
			$v$は位相速度．

	\subsection{Problem2.1(a)}
		\subsubsection{subsubsection name}
			時間に依存しない\sceq
			Time-Indpendent \sceq (TISE) \par
			TISE:
				\begin{equation}
					-\frac{\hbar^2}{2m}\frac{d^2 \psi(x)}{dx^2} + V(x)\psi(x) = E\psi(x)
				\end{equation}
			$\psi(x)$が規格化可能のとき，$E$は実数であることを示せ．
			(ちなみに今後，ポテンシャル$V(x)$は明示されない限り実数関数．
			（ちなみにちなみに$V$が複素関数になるのは特殊な研究をするときなどだけらしい）)

		\subsubsection{}
			TISEの両辺の左から$\psi^*$をかけて$-\infty 〜 \infty$で積分
			\begin{equation}
				-\frac{\hbar^2}{2m}\int^{\infty}_{-\infty}\psi^*
				\frac{d^2 \psi(x)}{dx^2} dx
				+ \int^{\infty}_{-\infty}V(x)\psi^*\psi(x) dx
				 = E\int^{\infty}_{-\infty} \psi^*\psi(x) dx
			\end{equation}

			今，$\psi$は規格化可能より，$\int^{\infty}_{-\infty}|\psi|^2 dx = 1$(=右辺).\par
			また，
				\begin{eqnarray}
					-\frac{\hbar^2}{2m}\int^{\infty}_{-\infty}\psi^*
					\frac{d^2 \psi(x)}{dx^2} dx \\
					& = & -\frac{\hbar^2}{2m} ([\psi^* \frac{d\psi}{dx}]^{\infty}_[-\infty] - \int^{\infty}_{-\infty}(\frac{d\psi^2}{dx})(\frac{d\psi}{dx})dx) \\
					& = & \frac{\hbar^2}{2m}\int^{\infty}_{-\infty}|\psi|^2 dx
				\end{eqnarray}
			よって，
			\begin{equation}
				E = \int^{\infty}_{-\infty}(\frac{\hbar^2}{2m}|\frac{d\psi}{dx}|^2 + V(x)|\psi|^2)dx
			\end{equation}
			（積分の中身の二項はそれぞれ実数）よって$E$は実数．
